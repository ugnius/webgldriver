282 /**
283  * Generates a unit vector of the same direction as the provided vec3
284  * If vector length is 0, returns [0, 0, 0]
285  *
286  * @param {vec3} vec vec3 to normalize
287  * @param {vec3} [dest] vec3 receiving operation result. If not specified result is written to vec
288  *
289  * @returns {vec3} dest if specified, vec otherwise
290  */
291 vec3.normalize = function (vec, dest) {
292     if (!dest) { dest = vec; }
293 
294     var x = vec[0], y = vec[1], z = vec[2],
295         len = Math.sqrt(x * x + y * y + z * z);
296 
297     if (!len) {
298         dest[0] = 0;
299         dest[1] = 0;
300         dest[2] = 0;
301         return dest;
302     } else if (len === 1) {
303         dest[0] = x;
304         dest[1] = y;
305         dest[2] = z;
306         return dest;
307     }
308 
309     len = 1 / len;
310     dest[0] = x * len;
311     dest[1] = y * len;
312     dest[2] = z * len;
313     return dest;
314 };

316 /**
317  * Generates the cross product of two vec3s
318  *
319  * @param {vec3} vec First operand
320  * @param {vec3} vec2 Second operand
321  * @param {vec3} [dest] vec3 receiving operation result. If not specified result is written to vec
322  *
323  * @returns {vec3} dest if specified, vec otherwise
324  */
325 vec3.cross = function (vec, vec2, dest) {
326     if (!dest) { dest = vec; }
327 
328     var x = vec[0], y = vec[1], z = vec[2],
329         x2 = vec2[0], y2 = vec2[1], z2 = vec2[2];
330 
331     dest[0] = y * z2 - z * y2;
332     dest[1] = z * x2 - x * z2;
333     dest[2] = x * y2 - y * x2;
334     return dest;
335 };

407     dest[2] = vec[2] + lerp * (vec2[2] - vec[2]);

429 /**
430  * Projects the specified vec3 from screen space into object space
431  * Based on the <a href="http://webcvs.freedesktop.org/mesa/Mesa/src/glu/mesa/project.c?revision=1.4&view=markup">Mesa gluUnProject implementation</a>
432  *
433  * @param {vec3} vec Screen-space vector to project
434  * @param {mat4} view View matrix
435  * @param {mat4} proj Projection matrix
436  * @param {vec4} viewport Viewport as given to gl.viewport [x, y, width, height]
437  * @param {vec3} [dest] vec3 receiving unprojected result. If not specified result is written to vec
438  *
439  * @returns {vec3} dest if specified, vec otherwise
440  */
441 vec3.unproject = function (vec, view, proj, viewport, dest) {
442     if (!dest) { dest = vec; }
443 
444     var m = mat4.create();
445     var v = new MatrixArray(4);
446     
447     v[0] = (vec[0] - viewport[0]) * 2.0 / viewport[2] - 1.0;
448     v[1] = (vec[1] - viewport[1]) * 2.0 / viewport[3] - 1.0;
449     v[2] = 2.0 * vec[2] - 1.0;
450     v[3] = 1.0;
451     
452     mat4.multiply(proj, view, m);
453     if(!mat4.inverse(m)) { return null; }
454     
455     mat4.multiplyVec4(m, v);
456     if(v[3] === 0.0) { return null; }
457 
458     dest[0] = v[0] / v[3];
459     dest[1] = v[1] / v[3];
460     dest[2] = v[2] / v[3];
461     
462     return dest;
463 };

1138         y *= len;

1423  * @param {vec3} up vec3 pointing "up"

1425  *
1426  * @returns {mat4} dest if specified, a new mat4 otherwise
1427  */
1428 mat4.lookAt = function (eye, center, up, dest) {
1429     if (!dest) { dest = mat4.create(); }
1430 
1431     var x0, x1, x2, y0, y1, y2, z0, z1, z2, len,
1432         eyex = eye[0],
1433         eyey = eye[1],
1434         eyez = eye[2],
1435         upx = up[0],
1436         upy = up[1],
1437         upz = up[2],
1438         centerx = center[0],
1439         centery = center[1],
1440         centerz = center[2];
1441

1443         return mat4.identity(dest);
1444     }
1445

1447     z0 = eyex - centerx;
1448     z1 = eyey - centery;
1449     z2 = eyez - centerz;
1450 
1451     // normalize (no check needed for 0 because of early return)
1452     len = 1 / Math.sqrt(z0 * z0 + z1 * z1 + z2 * z2);
1453     z0 *= len;
1454     z1 *= len;
1455     z2 *= len;
1456

1459     x1 = upz * z0 - upx * z2;
1460     x2 = upx * z1 - upy * z0;

1462     if (!len) {
1463         x0 = 0;
1464         x1 = 0;
1465         x2 = 0;
1466     } else {
1467         len = 1 / len;
1468         x0 *= len;
1469         x1 *= len;
1470         x2 *= len;
1471     }
1472 
1473     //vec3.normalize(vec3.cross(z, x, y));
1474     y0 = z1 * x2 - z2 * x1;

1476     y2 = z0 * x1 - z1 * x0;

1481         y1 = 0;
1482         y2 = 0;

1491     dest[1] = y0;
1492     dest[2] = z0;
1493     dest[3] = 0;
1494     dest[4] = x1;
1495     dest[5] = y1;
1496     dest[6] = z1;
1497     dest[7] = 0;
1498     dest[8] = x2;
1499     dest[9] = y2;
1500     dest[10] = z2;
1501     dest[11] = 0;

1943 /**
1944  * Performs a spherical linear interpolation between two quat4
1945  *
1946  * @param {quat4} quat First quaternion
1947  * @param {quat4} quat2 Second quaternion
1948  * @param {number} slerp Interpolation amount between the two inputs
1949  * @param {quat4} [dest] quat4 receiving operation result. If not specified result is written to quat
1950  *
1951  * @returns {quat4} dest if specified, quat otherwise
1952  */
1953 quat4.slerp = function (quat, quat2, slerp, dest) {
1954     if (!dest) { dest = quat; }
1955 
1956     var cosHalfTheta = quat[0] * quat2[0] + quat[1] * quat2[1] + quat[2] * quat2[2] + quat[3] * quat2[3],
1957         halfTheta,
1958         sinHalfTheta,
1959         ratioA,
1960         ratioB;
1961 
1962     if (Math.abs(cosHalfTheta) >= 1.0) {
1963         if (dest !== quat) {
1964             dest[0] = quat[0];
1965             dest[1] = quat[1];
1966             dest[2] = quat[2];
1967             dest[3] = quat[3];
1968         }
1969         return dest;
1970     }
1971 
1972     halfTheta = Math.acos(cosHalfTheta);
1973     sinHalfTheta = Math.sqrt(1.0 - cosHalfTheta * cosHalfTheta);
1974 
1975     if (Math.abs(sinHalfTheta) < 0.001) {
1976         dest[0] = (quat[0] * 0.5 + quat2[0] * 0.5);
1977         dest[1] = (quat[1] * 0.5 + quat2[1] * 0.5);
1978         dest[2] = (quat[2] * 0.5 + quat2[2] * 0.5);
1979         dest[3] = (quat[3] * 0.5 + quat2[3] * 0.5);
1980         return dest;
1981     }
1982 
1983     ratioA = Math.sin((1 - slerp) * halfTheta) / sinHalfTheta;
1984     ratioB = Math.sin(slerp * halfTheta) / sinHalfTheta;
1985 
1986     dest[0] = (quat[0] * ratioA + quat2[0] * ratioB);
1987     dest[1] = (quat[1] * ratioA + quat2[1] * ratioB);
1988     dest[2] = (quat[2] * ratioA + quat2[2] * ratioB);
1989     dest[3] = (quat[3] * ratioA + quat2[3] * ratioB);
1990 
1991     return dest;
1992 };
