138 BGAME.loadSettings = function() {
139 	if ( typeof localStorage === 'undefined' ) { return; }
140 	
141 	var set = String(localStorage.getItem('settings'));
142 	if ( set ) {
143 		set = window.JSON.parse(set);
144 		if ( set ) {
145 			if ( set.fieldOfView !== undefined ) { BGAME.fieldOfView = set.fieldOfView; }
146 			if ( set.camMode !== undefined ) { BGAME.camMode = set.camMode; }
147 			if ( set.audioGain !== undefined ) { BGAME.audioGain = set.audioGain; }
148 			if ( set.canvasScale !== undefined ) { BGAME.canvasScale = set.canvasScale; }
149 			if ( set.renderDis !== undefined ) { BGAME.renderDis = set.renderDis; }
150 		}
151 		
152 	}

156 BGAME.tick = function(time) {
157 	
158 	time = (new Date()).getTime();
159     
160     BGAME.input.handleInput();

162

168 			BGAME.activeScreen.render(BGAME.glc);

171

173 	//						+ 'RENDERED: ' + BGAME.rendered + '</br>';

176 	//						+ 'rframe: ' + BGAME.activeScreen.rframe + '<br/>';
177

179         if ( BGAME.help.style.display === 'none' ) {

196 BGAME.carControl = function(car, input, dt) {
197 	if ( input.isKeyDown(input.KEY_UP) ) {
198 		car.throttle += 0.2;
199 		if ( car.throttle > 1.0) {
200 			car.throttle = 1.0;
201 		}
202 	} else {
203 		if ( car.throttle <= 0.1 ) {
204 			car.throttle = 0;
205 		} else {
206 			car.throttle -= 0.1;
207 		}
208 	}
209 	
210 	if ( input.isKeyDown(input.KEY_DOWN) ) {
211 		car.brakes += 0.2;
212 		if ( car.brakes > 1.0) {
213 			car.brakes = 1.0;
214 		}
215 	} else {
216 		if ( car.brakes <= 0.2 ) {
217 			car.brakes = 0;
218 		} else {
219 			car.brakes -= 0.2;
220 		}
221 	}
222 	
223 	// steering
224 	if ( input.isKeyDown(input.KEY_LEFT) ) {
225 		car.steerAngle += 0.1;
226 	}
227 	if ( input.isKeyDown(input.KEY_RIGHT) ) {
228 		car.steerAngle -= 0.1;
229 	}
230 	if ( !input.isKeyDown(input.KEY_LEFT) && !input.isKeyDown(input.KEY_RIGHT) ) {
231 		if ( Math.abs(car.steerAngle) <= 0.1 ) {
232 			car.steerAngle = 0;
233 		} else {
234 			if ( car.steerAngle > 0 ) {
235 				car.steerAngle -= 0.1;
236 			} else {
237 				car.steerAngle += 0.1;
238 			}
239 		}
240 	}
241 	if ( car.steerAngle > 1 ) { car.steerAngle = 1; }
242 	if ( car.steerAngle < -1 ) { car.steerAngle = -1; }
243 	
244 	if ( input.wasKeyPressed(input.KEY_2) ) {
245 		car.gear++;
246 		if ( car.gear > 6 ) {car.gear = 6; }
247 	}
248 	if ( input.wasKeyPressed(input.KEY_1) ) {
249 		car.gear--;
250 		if ( car.gear < -1 ) { car.gear = -1; }
251 	}
252 	
253 	if ( input.gamepadAPI ) {	// Joystick controls
254 		if ( input.gamepadButton(6) > 0.02 || false) {
255 			car.brakes = input.gamepadButton(6);
256 		}
257 		if ( input.gamepadButton(7) > 0.02 || false) {
258 			car.throttle = input.gamepadButton(7);
259 		}
260 		
261 		if ( Math.abs(input.gamepadAxis(0)) > 0.05 || false) {
262 			car.steerAngle = -input.gamepadAxis(0);
263 		}
264 		
265 	}
266 };

270 BGAME.FFCamera = function(camera, input, dt, speed) {
271 	
272 	var vrot, hrot, posx, posy, posz;
273 	
274 	speed = speed || 100;
275 	dt = dt || 1/60;
276 	dt *= speed;
277 	
278 	// move player camera around	
279 	vrot = camera.rotation[0];
280 	hrot = camera.rotation[1]; //BGAME.hrot;
281 	posx = camera.position[0];
282 	posy = camera.position[1];
283 	posz = camera.position[2];	
284     
285     if ( input.isKeyDown(input.KEY_A) ) {
286         posx -= Math.cos(-hrot) * dt;
287         posz -= Math.sin(-hrot) * dt;
288     }
289     if ( input.isKeyDown(input.KEY_D) ) {
290         posx += Math.cos(-hrot) * dt;
291         posz += Math.sin(-hrot) * dt;
292     }
293     if ( input.isKeyDown(input.KEY_S) ) {
294         posx -= Math.sin(-hrot) * dt;
295         posz += Math.cos(-hrot) * dt;
296     }
297     if ( input.isKeyDown(input.KEY_W) ) {
298         posx += Math.sin(-hrot) * dt;
299         posz -= Math.cos(-hrot) * dt;
300     }
301 	if ( input.isKeyDown(input.KEY_Q) ) {
302         posy += dt;
303     }
304     if ( input.isKeyDown(input.KEY_Z) ) {
305         posy -= dt;
306     }
307     if ( input.isMouseLocked() || input.isButtonDown(input.BUTTON_RIGHT) ) {
308         hrot -= input.mousePositionDiffX() * 0.005;
309         vrot -= input.mousePositionDiffY() * 0.005;
310     }
311 	
312 	camera.setPosition(posx, posy, posz);
313     camera.setRotation(vrot, hrot, 0);
314 };

317 BGAME.FFCameraGamepad = function(camera, input, dt, speed) {
318 	
319 	var vrot, hrot, posx, posy, posz;
320 	
321 	speed = speed || 100;
322 	dt = dt || 1/60;
323 	dt *= speed;
324 	
325 	vrot = camera.rotation[0];
326 	hrot = camera.rotation[1]; //BGAME.hrot;
327 	posx = camera.position[0];
328 	posy = camera.position[1];
329 	posz = camera.position[2];
330 	
331 	var rx = input.gamepadAxis(input.PAD_AXIS_RX);
332 	var ry = input.gamepadAxis(input.PAD_AXIS_RY);
333 	var lx = input.gamepadAxis(input.PAD_AXIS_LX);
334 	var ly = input.gamepadAxis(input.PAD_AXIS_LY);
335 	
336 	lx = (KFC.clamp(Math.abs(lx)- 0.05, 0.0, 1.0) * KFC.sign(lx)) * dt * 0.5;
337 	ly = (KFC.clamp(Math.abs(ly)- 0.05, 0.0, 1.0) * KFC.sign(ly)) * dt * 0.5;
338 	
339 	posx -= Math.sin(-hrot) * ly - Math.cos(-hrot) * lx;
340 	posz += Math.cos(-hrot) * ly + Math.sin(-hrot) * lx;
341 	
342 	posy += (input.gamepadButton(input.PAD_RT) - input.gamepadButton(input.PAD_LT)) * 4.0;
343 	
344 	hrot -= (KFC.clamp(Math.abs(rx)- 0.1, 0.0, 1.0) * KFC.sign(rx)) * dt * 0.01;
345 	vrot -= (KFC.clamp(Math.abs(ry)- 0.1, 0.0, 1.0) * KFC.sign(ry)) * dt * 0.01;
346 	
347 	if ( vrot >  1.5 ) { vrot =  1.5; }
348 	if ( vrot < -1.5 ) { vrot = -1.5; }
349 	
350 	camera.setPosition(posx, posy, posz);
351     camera.setRotation(vrot, hrot, 0);
352 };
