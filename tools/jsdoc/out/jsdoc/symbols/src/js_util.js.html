 39 KFC.quat4 = function (w, v, norm, dest) {	// TODO, optimize array literals, ( check performance? )
 40 	if (norm !== true) { vec3.normalize(v); }
 41 	var r = w * 0.5;
 42 	var s = Math.sin(r);
 43 	
 44 	quat4.set([v[0]*s, v[1]*s, v[2]*s, Math.cos(r)], dest);
 45 };

 47 /**
 48  * Padds text in the front with pad character until text is long enought
 49  * @param {string} text Text to pad
 50  * @param {number} length Length to pad string to
 51  * @param {string} pad Padding character
 52  * @return {string} Padded text
 53  */
 54 KFC.padString = function( text, length, pad ) {
 55 	var l = text.length;
 56 	var n = KFC.clamp(length - l, 0, 256);
 57 	return (new Array(n+1)).join(pad[0]) + text;
 58 };

 71 KFC.vec3.sqrDis = function(vec1, vec2) {
 72 	return ((vec1[0]-vec2[0]) * (vec1[0]-vec2[0])) +
 73 			((vec1[1]-vec2[1]) * (vec1[1]-vec2[1])) +
 74 			((vec1[2]-vec2[2]) * (vec1[2]-vec2[2]));
 75 };

 77 /**
 78  * Extension to gl-matrix functions, adds scaled vector
 79  * @param {vec3} vec First operand
 80  * @param {vec3} vec2 Second operand
 81  * @param {number} val Scale value
 82  * @param {vec3} [dest] vec3 receiving operation result. If not specified result is written to vec
 83  * 
 84  * @returns {vec3} dest if specified, vec otherwise
 85  */
 86 KFC.vec3.addscaled = function (vec, vec2, val, dest) {
 87     if (!dest || vec === dest) {
 88         vec[0] += vec2[0] * val;
 89         vec[1] += vec2[1] * val;
 90         vec[2] += vec2[2] * val;
 91         return vec;
 92     }
 93 
 94     dest[0] = vec[0] + vec2[0] * val;
 95     dest[1] = vec[1] + vec2[1] * val;
 96     dest[2] = vec[2] + vec2[2] * val;
 97     return dest;
 98 };

101 KFC.loadFile = function (url, callback) {
102 	
103 	if ( typeof(exports) === 'undefined' ) {
104 		var xhr;
105 		xhr = new XMLHttpRequest();
106 		xhr.open("GET", url);
107 		xhr.onreadystatechange = function () {
108 			if (xhr.readyState === 4) {
109 				callback(xhr.responseText);
110 			}
111 		};
112 		xhr.send();
113 	} else {
114 		fs.readFile(url, 'utf-8', function(err, data) {
115 			callback(data);
116 		});
117 	}
118 };
