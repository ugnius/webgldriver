 63 KFC.Obj3D.prototype._makeMatrix = function () {
 64 	mat4.identity(this.mvMatrix);
 65 	mat4.translate(this.mvMatrix, this.pos);
 66 	mat4.multiply(this.mvMatrix, quat4.toMat4(this.rot));
 67 	
 68 	
 69 	mat4.toInverseMat3(this.mvMatrix, this.nMatrix);
 70 	mat3.transpose(this.nMatrix);
 71 	
 72 	this.dirty = false;
 73 };

 75 KFC.Obj3D.prototype.getMVMatrix = function () {
 76 	if ( this.dirty ) {
 77 		this._makeMatrix();
 78 	}
 79 	return this.mvMatrix;

186 KFC.LineMesh.prototype.apply = function () {
187 	var o = this.orgin,
188 		p = this.point,
189 		v = this.varr,
190 		gl;
191 		
192 	v[0] = o[0]; v[1] = o[1]; v[2] = o[2];
193 	if (this.add) {
194 		v[3] = o[0] + p[0];
195 		v[4] = o[1] + p[1];
196 		v[5] = o[2] + p[2];
197 	} else {
198 		v[3] = p[0]; v[4] = p[1]; v[5] = p[2];

201 	gl = this.gl;
202 	this.buffers.bVerPos = this.buffers.bVerPos || gl.createBuffer();
203 	gl.bindBuffer(gl.ARRAY_BUFFER, this.buffers.bVerPos);
204 	gl.bufferData(gl.ARRAY_BUFFER, v, gl.STREAM_DRAW);

294 KFC.ObjMesh.prototype._parse = function (text) {
295 	//console.log(text.length);
296 	
297 	var ln,	lp, n,
298 		v, v1, v2, v3, f, vt, vn, t,
299 		n1, n2, n3, t1, t2, t3, gl,
300 		va = [],
301 		fa = [],
302 		vta = [],
303 		vna = [],
304 		a = vec3.create(),
305 		b = vec3.create(),
306 		c = vec3.create(),
307 		nm = vec3.create();
308 		
309 	ln = text.split('\n');
310 	
311 	for ( n = 0; n < ln.length; n++ ) {
312 		if ( ln[n].length === 0 ) { continue; }
313 		lp = ln[n].split(' ');
314 		if ( lp.length === 0 ) { continue; }
315 		switch (lp[0]) {
316 			case "#" :
317 				//console.log('comment at: ' + n);
318 				break;
319 			case "o" :
320 				//console.log('Object: ' + lp[1]);
321 				break;
322 			case "v" :
323 				if ( lp.length === 4 ) {
324 					v = [parseFloat(lp[1]), parseFloat(lp[2]), parseFloat(lp[3])];
325 					if ( !isNaN(v[0]) && !isNaN(v[1]) && !isNaN(v[2]) ) {
326 						va.push(v);
327 					} else {
328 						console.error('Bad vertex data in ' + this.url_ + ':' + n);
329 					}
330 				} else {
331 					console.error('Wrong vertex data count in ' + this.url_ + ':' + n);
332 				}
333 				break;
334 			case "f" :
335 				if ( !this.firstFace ) {
336 					this.normals = vna.length > 0;
337 					this.texture = vta.length > 0;
338 					this.firstFace = true;
339 				}				
340 				
341 				if ( lp.length === 4 ) {	// this assumes only triangulated faces are listed ('f' f1 f2 f3)
342 					v1 = lp[1].split('/');
343 					v2 = lp[2].split('/');
344 					v3 = lp[3].split('/');
345 					f = [parseInt(v1[0], 10), parseInt(v2[0], 10), parseInt(v3[0], 10),
346 						 parseInt(v1[1], 10), parseInt(v2[1], 10), parseInt(v3[1], 10),
347 						 parseInt(v1[2], 10), parseInt(v2[2], 10), parseInt(v3[2], 10)];
348 					if ( !isNaN(f[0]) && !isNaN(f[1]) && !isNaN(f[2]) ) {
349 						fa.push(f);
350 					} else {
351 						console.error('Bad vertex data in ' + this.url_ + ':' + n);
352 					}
353 				} else {
354 					console.error('Wrong face data count in ' + this.url_ + ':' + n);
355 				}
356 				break;
357 			case 'vt' :
358 				if ( lp.length === 3 ) {
359 					vt = [parseFloat(lp[1]), parseFloat(lp[2])];
360 					if (  !isNaN(vt[0]) && !isNaN(vt[1]) ) {
361 						vta.push(vt);
362 					} else {
363 						console.error('Bad vertex texture data in ' + this.url_ + ':' + n);
364 					}
365 				} else {
366 					console.error('Bad vertex texture data in ' + this.url_ + ':' + n);
367 				}
368 				break;
369 			case 'vn' :
370 				if ( lp.length === 4 ) {
371 					vn = [parseFloat(lp[1]), parseFloat(lp[2]), parseFloat(lp[3])];
372 					if (  !isNaN(vn[0]) && !isNaN(vn[1]) && !isNaN(vn[2]) ) {
373 						vna.push(vn);
374 					} else {
375 						console.error('Bad vertex normal data in ' + this.url_ + ':' + n);
376 					}
377 				} else {
378 					console.error('Bad vertex normal data in ' + this.url_ + ':' + n);
379 				}
380 				break;
381 			default :
382 				//console.warn('Unknown obj data in ' + this.url_ + ':' + n + '\n' + ln[n]);
383 		}
384 	}
385 	
386 	var flip = -1;
387 	
388 	this.numTris = fa.length;
389 	
390 	this.taVerPos = new Float32Array(this.numTris*3*3);
391 	if ( this.normals === true ) {
392 		this.taNorm = new Float32Array(this.numTris*3*3);
393 	}
394 	if ( this.texture === true ) {
395 		this.taTexCoord	= new Float32Array(this.numTris*3*2);
396 	}
397 	
398 	if ( this.root === true ) {	// arrays needed to build tree
399 		this.taFaces = new Int16Array(this.numTris*3);
400 		this.taVerts = new Float32Array(va.length*3);
401 		this.taNorms = new Float32Array(this.numTris*3);
402 		
403 		for ( t = 0; t < this.numTris; t++ ) {
404 			this.taFaces[t*3    ] = fa[t][0] - 1;
405 			this.taFaces[t*3 + 1] = fa[t][1] - 1;
406 			this.taFaces[t*3 + 2] = fa[t][2] - 1;
407 		}
408 		
409 		for ( t = va.length - 1; t !== -1; t -= 1 ) {
410 			this.taVerts[t*3    ] = va[t][0];
411 			this.taVerts[t*3 + 1] = va[t][1];
412 			this.taVerts[t*3 + 2] = va[t][2];
413 		}
414 		
415 		for ( t = 0; t < this.numTris; t++ ) {
416 			// vertices
417 			v1 = va[fa[t][0]-1];
418 			v2 = va[fa[t][1]-1];
419 			v3 = va[fa[t][2]-1];
420 			
421 			// build normals from vertice data for flat collison surfaces
422 			vec3.subtract(v1, v2, a);
423 			vec3.subtract(v2, v3, b);
424 			vec3.cross(a, b, nm);
425 			vec3.normalize(nm);
426 			
427 			this.taNorms[t*3 + 0] = nm[0]; 
428 			this.taNorms[t*3 + 1] = nm[1];
429 			this.taNorms[t*3 + 2] = nm[2];
430 		}
431 		
432 	}
433 	
434 	
435 	for (t = 0; t < this.numTris; t++ ) {
436 		//var n = i*3;
437 		
438 		// vertices
439 		v1 = va[fa[t][0]-1];
440 		v2 = va[fa[t][1]-1];
441 		v3 = va[fa[t][2]-1];
442 		
443 		this.taVerPos[t*3*3    ] = v1[0];
444 		this.taVerPos[t*3*3 + 1] = v1[1];
445 		this.taVerPos[t*3*3 + 2] = v1[2];
446 		this.taVerPos[t*3*3 + 3] = v2[0];
447 		this.taVerPos[t*3*3 + 4] = v2[1];
448 		this.taVerPos[t*3*3 + 5] = v2[2];
449 		this.taVerPos[t*3*3 + 6] = v3[0];
450 		this.taVerPos[t*3*3 + 7] = v3[1];
451 		this.taVerPos[t*3*3 + 8] = v3[2];
452 		
453 		// Normals		
454 		if ( this.normals ) {
455 	
456 			n1 = vna[fa[t][6]-1];
457 			n2 = vna[fa[t][7]-1];
458 			n3 = vna[fa[t][8]-1];
459 			
460 			this.taNorm[t*3*3    ] = n1[0];
461 			this.taNorm[t*3*3 + 1] = n1[1];
462 			this.taNorm[t*3*3 + 2] = n1[2];
463 			this.taNorm[t*3*3 + 3] = n2[0];
464 			this.taNorm[t*3*3 + 4] = n2[1];
465 			this.taNorm[t*3*3 + 5] = n2[2];
466 			this.taNorm[t*3*3 + 6] = n3[0];
467 			this.taNorm[t*3*3 + 7] = n3[1];
468 			this.taNorm[t*3*3 + 8] = n3[2];
469 		}
470 
471 		// Texture coordinates
472 		if ( this.texture ) {
473 			
474 			t1 = vta[fa[t][3]-1];
475 			t2 = vta[fa[t][4]-1];
476 			t3 = vta[fa[t][5]-1];
477 			
478 			this.taTexCoord[t*3*2    ] = t1[0]; // same normal for all triangle vertices
479 			this.taTexCoord[t*3*2 + 1] = t1[1];
480 			this.taTexCoord[t*3*2 + 2] = t2[0];
481 			this.taTexCoord[t*3*2 + 3] = t2[1];
482 			this.taTexCoord[t*3*2 + 4] = t3[0];
483 			this.taTexCoord[t*3*2 + 5] = t3[1];
484 		}
485 	}
486 	
487 	if ( typeof(exports) === 'undefined' ) {	// exclude from node.js
488 		gl = this.gl;
489 		
490 		this.buffers.bVerPos = gl.createBuffer();
491 		gl.bindBuffer(gl.ARRAY_BUFFER, this.buffers.bVerPos);
492 		gl.bufferData(gl.ARRAY_BUFFER, this.taVerPos, gl.STATIC_DRAW);
493 		
494 		if ( this.normals ) {
495 			this.buffers.bNorm = gl.createBuffer();
496 			gl.bindBuffer(gl.ARRAY_BUFFER, this.buffers.bNorm);
497 			gl.bufferData(gl.ARRAY_BUFFER, this.taNorm, gl.STATIC_DRAW);
498 		}
499 		
500 		if ( this.texture ) {
501 			this.buffers.bTexCoord = gl.createBuffer();
502 			gl.bindBuffer(gl.ARRAY_BUFFER, this.buffers.bTexCoord);
503 			gl.bufferData(gl.ARRAY_BUFFER, this.taTexCoord, gl.STATIC_DRAW);
504 		}
505 	}
506 	
507 	var mesh = this;
508 	
509 	if ( this.root ) {
510 		setTimeout(function (){	// call this later, to not freeze stuff too much TODO: move to worker thread
511 			
512 			var t;
513 			mesh.root = new KFC.BTree(mesh);
514 			
515 			for ( t = 0; t < mesh.numTris; t++ ) {
516 				mesh.root.add(t);
517 			}
518 			mesh.root.build(0);
519 			
520 			if ( mesh.cb_ ) {
521 				mesh.cb_();
522 				mesh.cb_ = null;
523 			}
524 			
525 		}, 1);
526 	} else {
527 		if ( this.cb_ ) {
528 			this.cb_();
529 			this.cb_ = null;
530 		}
531 	}
532 	
533 	this.ready = true;
534 	
535 };

537 KFC.ObjMesh.prototype._render = function (gl, shader, object) {
538 	
539 	shader.attribPointer('aVerPosition', this.buffers.bVerPos, 3);
540 	
541 	var fix = 'aVerNormal';
542 	
543 	if ( shader.attributes[fix] !== undefined ) {	// if shader does not require normal attribute
544 		if ( this.normals === true ) {
545 			shader.enableAttrib('aVerNormal');			
546 			shader.attribPointer('aVerNormal', this.buffers.bNorm, 3);
547 		} else {
548 			shader.disableAttrib('aVerNormal');
549 		}
550 	}	
551 	
552 	if ( this.texture && object.tex0 ) {
553 		shader.enableAttrib('aTexCoord');
554 		shader.attribPointer('aTexCoord', this.buffers.bTexCoord, 2);
555 		
556 		shader.bindTexture(0, object.tex0.tex);		
557 		if ( object.tex1 ) {
558 			shader.bindTexture(1, object.tex1.tex);
559 		}
560 		if ( object.tex2 ) {
561 			shader.bindTexture(2, object.tex2.tex);
562 		}
563 		if ( object.tex3 ) {
564 			shader.bindTexture(3, object.tex3.tex);
565 		}
566 	}	
567 	
568 	gl.drawArrays(gl.TRIANGLES, 0, this.numTris*3);
569 	
570 };

609 		v, v1, v2, v3, f, vt, vn, t,
610 		lastoff = '',
611 		n1, n2, n3, t1, t2, t3, gl,
612 		va = [],
613 		fa = [],
614 		vta = [],
615 		vna = [];
616 		
617 	ln = text.split('\n');
618 	
619 	for ( n = 0; n < ln.length; n++ ) {
620 		if ( ln[n].length === 0 ) { continue; }
621 		lp = ln[n].split(' ');
622 		if ( lp.length === 0 ) { continue; }
623 		switch (lp[0]) {
624 			case "#" :
625 				//console.log('comment at: ' + n);
626 				break;
627 			case "o" :
628 				//console.log('Object: ' + lp[1]);
629 				break;
630 			case "g" :
631 				//console.log('Object: ', (lp[1])[0], fa.length);
632 				this.offsets[lp[1][0]] = fa.length;
633 				this.size[lastoff] = fa.length - this.offsets[lastoff];
634 				lastoff = lp[1][0];
635 				
636 				break;
637 			case "v" :
638 				if ( lp.length === 4 ) {
639 					v = [parseFloat(lp[1]), parseFloat(lp[2]), parseFloat(lp[3])];
640 					if ( !isNaN(v[0]) && !isNaN(v[1]) && !isNaN(v[2]) ) {
641 						va.push(v);
642 					} else {
643 						console.error('Bad vertex data in ' + this.url_ + ':' + n);
644 					}
645 				} else {
646 					console.error('Wrong vertex data count in ' + this.url_ + ':' + n);
647 				}
648 				break;
649 			case 'vn' :
650 				if ( lp.length === 4 ) {
651 					vn = [parseFloat(lp[1]), parseFloat(lp[2]), parseFloat(lp[3])];
652 					if (  !isNaN(vn[0]) && !isNaN(vn[1]) && !isNaN(vn[2]) ) {
653 						vna.push(vn);
654 					} else {
655 						console.error('Bad vertex normal data in ' + this.url_ + ':' + n);
656 					}
657 				} else {
658 					console.error('Bad vertex normal data in ' + this.url_ + ':' + n);
659 				}
660 				break;
661 			case "f" :
662 				if ( lp.length === 4 ) {	// this assumes only triangulated faces are listed ('f' f1 f2 f3)
663 					v1 = lp[1].split('/');
664 					v2 = lp[2].split('/');
665 					v3 = lp[3].split('/');
666 					f = [parseInt(v1[0], 10), parseInt(v2[0], 10), parseInt(v3[0], 10),
667 						 parseInt(v1[1], 10), parseInt(v2[1], 10), parseInt(v3[1], 10),
668 						 parseInt(v1[2], 10), parseInt(v2[2], 10), parseInt(v3[2], 10)];
669 					if ( !isNaN(f[0]) && !isNaN(f[1]) && !isNaN(f[2]) ) {
670 						fa.push(f);
671 					} else {
672 						console.error('Bad vertex data in ' + this.url_ + ':' + n);
673 					}
674 				} else {
675 					console.error('Wrong face data count in ' + this.url_ + ':' + n);
676 				}
677 				break;
678 			default :
679 				//console.warn('Unknown obj data in ' + this.url_ + ':' + n + '\n' + ln[n]);
680 		}
681 	}
682 	
683 	this.size[lastoff] = fa.length - this.offsets[lastoff];
684 	
685 	var flip = -1;
686 	
687 	this.numTris = fa.length;
688 	
689 	this.taVerPos	= new Float32Array(this.numTris*3*3);
690 	this.taNorm		= new Float32Array(this.numTris*3*3);	
691 	
692 	for (t = 0; t < this.numTris; t++ ) {
693 		//var n = i*3;
694 		
695 		// vertices
696 		v1 = va[fa[t][0]-1];
697 		v2 = va[fa[t][1]-1];
698 		v3 = va[fa[t][2]-1];
699 		
700 		this.taVerPos[t*3*3    ] = v1[0];
701 		this.taVerPos[t*3*3 + 1] = v1[1];
702 		this.taVerPos[t*3*3 + 2] = v1[2];
703 		this.taVerPos[t*3*3 + 3] = v2[0];
704 		this.taVerPos[t*3*3 + 4] = v2[1];
705 		this.taVerPos[t*3*3 + 5] = v2[2];
706 		this.taVerPos[t*3*3 + 6] = v3[0];
707 		this.taVerPos[t*3*3 + 7] = v3[1];
708 		this.taVerPos[t*3*3 + 8] = v3[2];
709 		
710 		// Normals		
711 		n1 = vna[fa[t][6]-1];
712 		n2 = vna[fa[t][7]-1];
713 		n3 = vna[fa[t][8]-1];
714 		
715 		this.taNorm[t*3*3    ] = -n1[0]; // same normal for all triangle vertices
716 		this.taNorm[t*3*3 + 1] = -n1[1];
717 		this.taNorm[t*3*3 + 2] = -n1[2];
718 		this.taNorm[t*3*3 + 3] = -n2[0];
719 		this.taNorm[t*3*3 + 4] = -n2[1];
720 		this.taNorm[t*3*3 + 5] = -n2[2];
721 		this.taNorm[t*3*3 + 6] = -n3[0];
722 		this.taNorm[t*3*3 + 7] = -n3[1];
723 		this.taNorm[t*3*3 + 8] = -n3[2];
724 	}
725 	
726 	gl = this.gl;
727 	
728 	this.buffers.bVerPos = gl.createBuffer();
729 	gl.bindBuffer(gl.ARRAY_BUFFER, this.buffers.bVerPos);
730 	gl.bufferData(gl.ARRAY_BUFFER, this.taVerPos, gl.STATIC_DRAW);
731 	
732 	this.buffers.bNorm = gl.createBuffer();
733 	gl.bindBuffer(gl.ARRAY_BUFFER, this.buffers.bNorm);
734 	gl.bufferData(gl.ARRAY_BUFFER, this.taNorm, gl.STATIC_DRAW);
735 	
736 	
737 	this.ready = true;
738 	
739 	if ( typeof(this._cb) == 'function' ) {
740 		this._cb();
741 		this._cb = null;
742 	}
743 };

746 KFC.TextMesh.prototype._render = function (gl, shader, object) {
747 	
748 	if ( typeof(object.text) !== 'string' ) { throw Error('TextMesh objects must have a text string!'); }
749 	
750 	var uniLoc = shader.uniLocs[shader.uniIds.indexOf(KFC.UniId.uOffset)];
751 	var i;
752 
753 	shader.enableAttrib('aVerPosition');
754 	shader.attribPointer('aVerPosition', this.buffers.bVerPos, 3);
755 	
756 	shader.enableAttrib('aVerNormal');
757 	shader.attribPointer('aVerNormal', this.buffers.bNorm, 3);
758 	
759 	for ( i = 0; i < object.text.length; i++ ) {
760 		var c = object.text[i].toLowerCase();
761 		if ( c === ' ' ) { continue; }
762 		if ( this.offsets[c] === undefined ) { c = '_'; }
763 		gl.uniform1i(uniLoc, i);
764 		gl.drawArrays(gl.TRIANGLES, this.offsets[c]*3, this.size[c]*3);
765 	}
766 	
767 };
