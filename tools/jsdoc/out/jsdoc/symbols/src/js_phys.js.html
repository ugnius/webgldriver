 40 KFC.BTree.prototype.seeBounds = function() {
 41 	
 42 	var i, t, v, p, a;
 43 	
 44 	// reset
 45 	this.min[0] =  1e+10;
 46 	this.min[1] =  1e+10;
 47 	this.min[2] =  1e+10;
 48 	this.max[0] = -1e+10;
 49 	this.max[1] = -1e+10;
 50 	this.max[2] = -1e+10;
 51 	
 52 	for (i = 0; i < this.ta.length; i++) {
 53 		t = this.ta[i] * 3;
 54 		for (v = 0; v < 3; v++) {
 55 			
 56 			for (a = 0; a < 3; a++) {
 57 				p = this.mesh.taVerts[this.mesh.taFaces[t+v]*3 + a];
 58 				if ( p > this.max[a] ) { this.max[a] = p; }
 59 				if ( p < this.min[a] ) { this.min[a] = p; }
 60 			}
 61 		}
 62 	}
 63 	
 64 };

 67 KFC.BTree.prototype.build = function(lvl) {
 68 	var split, mid, t, p, i, a, na, q;
 69 	this.seeBounds();
 70 	
 71 	this.level = lvl;
 72 	
 73 	if ( this.ta.length === 0 ) { console.log(0); }
 74 	
 75 	if ( this.ta.length > 2 && lvl < 50 ) {
 76 		split = (((this.max[0] - this.min[0]) > (this.max[2] - this.min[2]))?0:2);
 77 		
 78 		this.split = split;
 79 		
 80 		mid = ((split) ? (this.max[2] + this.min[2])/2 : (this.max[0] + this.min[0])/2);
 81 		  q =  (split) ? (this.max[2] - this.min[2])/4 : (this.max[0] - this.min[0])/4;
 82 		
 83 		this.mid = mid;
 84 		
 85 		this.z = new KFC.BTree(this.mesh);
 86 		this.o = new KFC.BTree(this.mesh);
 87 		
 88 		na = [];
 89 		
 90 		while ((t = this.ta.pop()) !== undefined) {
 91 			// find max and min coordinate in split axis
 92 			i = a = this.mesh.taVerts[this.mesh.taFaces[t*3]*3+split];
 93 			p = this.mesh.taVerts[this.mesh.taFaces[t*3+1]*3+split];
 94 			if ( p < i ) { i = p; }
 95 			if ( p > a ) { a = p; }
 96 			p = this.mesh.taVerts[this.mesh.taFaces[t*3+2]*3+split];
 97 			if ( p < i ) { i = p; }
 98 			if ( p > a ) { a = p; }
 99 			
100 			if ( (mid-i) > q && (a-mid) > q ) {
101 				na.push(t);
102 			} else {
103 				if ( (mid-i) > (a-mid) ) {
104 					this.o.add(t);
105 				} else {
106 					this.z.add(t);
107 				}
108 			}
109 		}
110 		
111 		this.ta = na;
112 		
113 		if ( this.z.ta.length === 0 ) { this.z = null; }
114 		else { this.z.build(lvl+1); }
115 		
116 		if ( this.o.ta.length === 0 ) { this.o = null; }
117 		else { this.o.build(lvl+1); }
118 		
119 	}
120 	
121 };

123 // Finds highest mesh position at line
124 KFC.BTree.prototype.yAt = function(x, z) {
125 	if ( x < this.min[0] || x > this.max[0] || z > this.max[2] || z < this.min[2] ) { return false; }
126 	
127 	var ret;
128 	var max = -1e2;
129 	var n = [0,1,0];
130 	var i;
131 	
132 	for (i = this.ta.length-1; i >= 0; i--) {
133 		ret = this.testY(x, z, this.ta[i]);
134 		if ( ret !== false && ret.r > max ) {
135 			max = ret.r; n = ret.n;
136 		}
137 	}
138 	
139 	if ( this.z !== null ) {
140 		ret = this.z.yAt(x, z);
141 		if ( ret !== false && ret.r > max ) {
142 			max = ret.r; n = ret.n;
143 		}
144 	}
145 	
146 	if ( this.o !== null ) {
147 		ret = this.o.yAt(x, z);
148 		if ( ret !== false && ret.r > max ) {
149 			max = ret.r; n = ret.n;
150 		}
151 	}
152 	
153 	return {'r':max, 'n':n};
154 };

158 KFC.BTree.prototype.testY = function(x, z, t) {
159 	var a, b, c, i, m, A, B, C,
160 		t3 = t+t+t;
161 		
162 	m = this.mesh;
163 	
164 	i = m.taFaces[t3]*3;	a = [m.taVerts[i++], m.taVerts[i++], m.taVerts[i]];
165 	i = m.taFaces[t3+1]*3;	b = [m.taVerts[i++], m.taVerts[i++], m.taVerts[i]];
166 	if ( KFC.dot2d(x-a[0], z-a[2], a[2]-b[2], b[0]-a[0]) > 0 ) return false;
167 	i = m.taFaces[t3+2]*3;	c = [m.taVerts[i++], m.taVerts[i++], m.taVerts[i]];
168     if ( KFC.dot2d(x-b[0], z-b[2], b[2]-c[2], c[0]-b[0]) > 0 ) return false;
169     if ( KFC.dot2d(x-c[0], z-c[2], c[2]-a[2], a[0]-c[0]) > 0 ) return false;
170 	
171 	A = m.taNorms[t3++];
172 	B = m.taNorms[t3++];
173 	C = m.taNorms[t3];
174 	
175     if ( A === 0 && B === 0 && C === 0 ) {
176         return false;
177 	}
178         
179     var mD = A*a[0] + B*a[1] + C*a[2];
180     
181     return {'r':(A*x + C*z - mD) / (-B), 'n':[A, B, C]};
182 };

184 /**
185  * Tests ray collision against mesh tree
186  * @param {!vec3} org Ray orgin 
187  * @param {!vec3} dir Ray direction, should be normalized
188  * @param {number=} opt_max Maximum ray length to test
189  */
190 KFC.BTree.prototype.rayTest = function( org, dir, opt_max ) {
191 	
192 	if ( opt_max !== undefined ) { this.mesh.dis = opt_max; }
193 	else { opt_max = this.mesh.dis; }
194 	
195 	var i;
196 	
197 	var tfar	=  100000.1,
198 		tnear	= -100000.1,
199 		t1, t2;
200 	
201 	for ( i = 2; i >= 0; i-- ) {
202 		if ( dir[i] === 0 ) {
203 			if ( org[i] < this.min[i] || org[i] > this.max[i] ) {
204 				return false;
205 			}
206 			continue;
207 		}
208 		
209 		t1 = ( this.min[i] - org[i] ) / dir[i];
210 		t2 = ( this.max[i] - org[i] ) / dir[i];
211 		
212 		if ( t1 > t2 ) {
213 			if ( t2 > tnear ) { tnear = t2; }
214 			if ( t1 < tfar  ) { tfar  = t1; }
215 		} else {
216 			if ( t1 > tnear ) { tnear = t1; }
217 			if ( t2 < tfar  ) { tfar  = t2; }
218 		}
219 		
220 		if ( tnear > tfar ) { return false; }
221 		if ( tfar <= 0 ) { return false; }
222 		if ( tnear > opt_max ) { return false; }
223 	}
224 	
225 	for ( i = this.ta.length-1; i >= 0; i-- ) {
226 		this.rayTri( org, dir, this.ta[i]);
227 	}
228 	
229 	if ( this.z !== null ) {
230 		this.z.rayTest(org, dir);
231 	}
232 	
233 	if ( this.o !== null ) {
234 		this.o.rayTest(org, dir);
235 	}
236 	
237 	if ( this.mesh.dis === opt_max ) { return false; }
238 	
239 	return true;
240 	
241 };

244 KFC.BTree.prototype.rayTri = function( org, dir, t ) {
245 	
246 	var t3 = t+t+t;
247 	var i, m, A, B, C;
248 	
249 	m = this.mesh;
250 	
251 	var cp	= m.tcp,	// collision point
252 		ctc	= m.ctc,
253 		cr	= m.cr,
254 		tn	= m.tn,
255 		a	= m.a,
256 		b	= m.b,
257 		c	= m.c;
258 	
259 	i = m.taFaces[t3]*3;	a[0] = m.taVerts[i++]; a[1] = m.taVerts[i++]; a[2] = m.taVerts[i];
260 	i = m.taFaces[t3+1]*3;	b[0] = m.taVerts[i++]; b[1] = m.taVerts[i++]; b[2] = m.taVerts[i];
261 	i = m.taFaces[t3+2]*3;	c[0] = m.taVerts[i++]; c[1] = m.taVerts[i++]; c[2] = m.taVerts[i];
262 	
263 	tn[0] = m.taNorms[t3++];
264 	tn[1] = m.taNorms[t3++];
265 	tn[2] = m.taNorms[t3];
266 	
267 	
268 	var D = -1 * vec3.dot(tn, a);
269 	var M = vec3.dot(tn, dir);
270 	if ( M === 0 ) { return false; }
271 	
272 	var dis = -1 * (vec3.dot(org, tn) + D) / M;
273 	
274 	if ( dis >= m.dis ) { return false; }
275 	if ( dis < 0 ) { return false; }
276 	
277     vec3.scale(dir, dis, cp );
278     vec3.add(cp, org);
279 	
280 	// is collision point in triangle, TODO: make functions?
281     vec3.subtract(a, b, cr);
282     vec3.cross(cr, tn);
283     vec3.subtract(cp, a, ctc);
284     if ( vec3.dot(cr, ctc) < 0 ) { return false; }
285 	
286     vec3.subtract(b, c, cr);
287     vec3.cross(cr, tn);
288     vec3.subtract(cp, b, ctc);
289     if ( vec3.dot(cr, ctc) < 0 ) { return false; }
290 	
291     vec3.subtract(c, a, cr);
292     vec3.cross(cr, tn);
293     vec3.subtract(cp, c, ctc);
294     if ( vec3.dot(cr, ctc) < 0 ) { return false; }
295 
296 	// set new values to mesh
297 	vec3.set(cp, m.cp);
298 	vec3.set(tn, m.nv);
299 	
300 	m.dis = dis;
301 	
302 	return true;
303 };
