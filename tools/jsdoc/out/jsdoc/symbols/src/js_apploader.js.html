 74 /**
 75  * Checks for supported APIs
 76  */
 77 APP.checkAPISupport = function() {
 78     
 79     var error = false,
 80 		canvas, gl;
 81     
 82     // WebGl
 83     if (APP.webGL_req) {
 84 		canvas = document.createElement('canvas');
 85 		gl = canvas.getContext('experimental-webgl') || canvas.getContext('webgl');
 86         if ( gl ) {
 87             APP.log('WebGL is supported', 'success');
 88         } else {
 89             APP.log('WebGL is not supported!', 'error');
 90             error = true;
 91         }
 92     }
 93     
 94     // WebSockets
 95     if (APP.webSockets_req) {
 96         if ( window.MozWebSocket || window.WebSocket ) {
 97             APP.log('WebSockets are supported', 'success');
 98         } else {
 99             APP.log('WebSockets are not supported!', 'error');
100             error = true;
101         }
102 	}
103     
104     // Mouse Lock API
105     if (APP.mouseLockAPI_req) {
106         if ( navigator.pointer || navigator.webkitPointer ) {
107             APP.log('Mouse Lock API is supported', 'success');
108         } else {
109             APP.log('Mouse Lock API is not supported!', 'warning');
110             //error = true;
111         }
112     }
113 	
114 	// Gamepad API
115     if (APP.gamepadAPI_req) {
116         if ( navigator.gamepads || navigator.webkitGamepads ) {
117             APP.log('Gamepad API is supported', 'success');
118         } else {
119             APP.log('Gamepad API is not supported!', 'warning');
120             //error = true;
121         }
122     }
123 	
124 	// Web Audio API
125     if (APP.webAudioAPI_req) {
126         if ( window.AudioContext || window.webkitAudioContext ) {
127             APP.log('Web Audio API is supported', 'success');
128         } else {
129             APP.log('Web Audio API is not supported!', 'warning');
130             //error = true;
131         }
132     }
133 	
134     
135     if (!error) {
136 		APP.log('Loading application...');
137 		if ( COMPILED === true ) {
138             BGAME.main();
139 		} else {
140 			APP.loadApplicationFiles();
141 		}
142     } else {
143 		APP.log('Sorry, this game will not work here =/');
144 	}
145     
146 };

149 /**
150  * Loads files in APP.scripts array
151  */
152 APP.loadApplicationFiles = function() {
153     var count = APP.scripts.length,
154 		cb, s;
155     
156     cb = function(url){		
157 		count--;
158         //APP.log('<p>' + url + ' loaded. '  + count + ' left</p>');
159         APP.add('.');
160         if ( count === 0 ) {
161             window[APP.start_obj][APP.start_func]();
162         }
163     };
164 	
165 	while ( (s = APP.scripts.pop()) !== undefined ) {
166         APP.loadScript(s, cb);
167 	}
168 };

171 /**
172  * Adds script tag to html header
173  * @param {string} url Script url
174  * @param {function(string)} callback function to call when script is loaded
175  */
176 APP.loadScript = function(url, callback) {
177 	/** @type {Element} */
178     var script = document.createElement('script');
179     script.type = 'text/javascript';
180     script.src = url;
181     script.onload = function(){
182         var lurl = url;
183         callback(lurl);
184     };
185     document.getElementsByTagName('head')[0].appendChild(script);
186 };

188 /**
189  * Post AJAX message to server
190  * @param {string} type Type of message. Should be 'err' or 'log'
191  * @param {string} msg Message content.
192  */
193 APP.postMessage = function(type, msg) {
194 	/** @type {XMLHttpRequest} */
195 	var xhr = new XMLHttpRequest();
196 	xhr.open('POST', 'post.php');
197 	xhr.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
198 	xhr.send('data=' + type + '&msg=' + msg);
199 };

201 /**
202  * Global error handler.
203  * @param {string} errorMsg Error message
204  * @param {string} url file in which error has occured
205  * @param {number} lineNumber Error line number
206  */
207 APP.onError = function(errorMsg, url, lineNumber) {
208 	
209 	/** @type {string} */
210 	var str;
211 	/** @type {string} */
212 	var msg;
213 	/** @type {Array.<string>} */
214 	var strp;
215 	
216 	if ( APP.in_error === true ) { return; }	// to avoid infinite loop
217 	APP.in_error = true;
218 	
219 	strp = url.split('/');
220 	str = strp[strp.length-1];
221 	msg = errorMsg + '  ' + str + ':' + lineNumber;
222 	
223 	APP.log( '<p class="error">' + msg + '</p>');
224 	APP.postMessage('err', msg);
225 	
226 	APP.in_error = false;
227 };
