 18 /**
 19  * WebGL context and redering class
 20  * @constructor
 21  * @param {HTMLElement} canvas
 22  * @param {Object.<string, *>} attributes paramters to pass to WebGL
 23  */
 24 KFC.CGLC = function (canvas, attributes) {
 25 	
 26 	if ( !canvas || typeof(canvas.getContext) !== 'function' ) { this.gl = null; return; }
 27 	
 28 	try {
 29 		this.gl = canvas.getContext('experimental-webgl', attributes) || canvas.getContext('webgl', attributes);
 30 	} catch (e) {}
 31 	if (!this.gl) {
 32 		console.error('cannot get webgl context');
 33 		return;
 34 	}
 35 	
 36 	this.canvas = canvas;
 37 	
 38 	this.ready = false;
 39 
 40 	this._init();
 41 	
 42 };

 44 /**
 45  * CGLC initalization function
 46  */
 47 KFC.CGLC.prototype._init = function () {
 48 	this.activeShader = null;
 49 	this.shaders = {};
 50 	this.textures = {};
 51 	this.meshes = [];
 52 	this.framebuffers = {};
 53 	
 54 	//this.gl.clearColor(0.14, 0.14, 0.14, 1.0);
 55 	this.gl.clearColor(1.0, 1.0, 1.0, 1.0);
 56 	this.gl.enable(this.gl.DEPTH_TEST);		
 57 	this.gl.blendFunc(this.gl.SRC_ALPHA, this.gl.ONE_MINUS_SRC_ALPHA);
 58 	this.gl.enable(this.gl.CULL_FACE);
 59 	
 60 	this.width = this.canvas.width;
 61 	this.height = this.canvas.height;
 62 	this.fbResize = false;
 63 	
 64 	this.culling = true;
 65 	this.blending = false;
 66 	
 67 	this.lastBuffer = null;
 68 	this.activeTexture = 0;
 69 	this.bindedTextures = [];
 70 	
 71 	var taVerPos;
 72 	taVerPos = new Float32Array([ // vertex positions array
 73 		-1,  1,  0,  -1, -1,  0,   1, -1,  0,
 74 		 1, -1,  0,   1,  1,  0,  -1,  1,  0 ]);	
 75 	this.bScreenVerPos = this.gl.createBuffer();
 76 	this.gl.bindBuffer(this.gl.ARRAY_BUFFER, this.bScreenVerPos);
 77 	this.gl.bufferData(this.gl.ARRAY_BUFFER, taVerPos, this.gl.STATIC_DRAW);
 78 	
 79 	//console.log(this.gl.getSupportedExtensions());
 80 	this.gl.getExtension('OES_texture_float'); 
 81 	
 82 	this.ready = true;
 83 };

 85 /**
 86  * Releases resources allocated on GPU, should be called before closing page
 87  */
 88 KFC.CGLC.prototype.unload = function () {
 89 	
 90 	var gl = this.gl,
 91 		n, fb;
 92 	
 93 	for ( n in this.framebuffers ) {
 94 		if ( this.framebuffers.hasOwnProperty(n) ) {
 95 			fb = this.framebuffers[n];
 96 			
 97 			gl.deleteFramebuffer(fb.fb);
 98 			gl.deleteTexture(fb.tex);
 99 			gl.deleteRenderbuffer(fb.rb);
100 		}
101 	}
102 	
103 };

120 /**
121  * Renders all objects in a scene from specified camera
122  * @param {KFC.Scene} scene Scene to render objects from
123  * @param {KFC.Camera} camera Camera to get perspective matrix from
124  */
125 KFC.CGLC.prototype.render = function (scene, camera) {
126 	
127 	var gl = this.gl,
128 		o, i;
129 	
130 	if (scene.autoclear) {
131 		gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
132 	}
133 	
134 	for (i = 0; i < scene.objects.length; i++) {
135 		o = scene.objects[i];		
136 		this.renderObject(scene, camera, o, i);		
137 	}
138 	
139 	for (i = 0; i < scene.endObjects.length; i++) {		
140 		o = scene.endObjects[i];		
141 		this.renderObject(scene, camera, o, i);
142 	}
143 	
144 };

146 /**
147  * Renders single object
148  * @param {KFC.Scene} scene Scene object is from
149  * @param {KFC.Camera} camera Camera to get perspective matrix from
150  * @param {KFC.Obj3D} o Object to render
151  * @param {number} i Object id in a scene list
152  */
153 KFC.CGLC.prototype.renderObject = function ( scene, camera, o, i ) {
154 	
155 	// 
156 	var gl = this.gl,
157 		u, l, v, j;
158 	
159 	if ( !o.draw || !o.mesh.ready || !o.shader.ready ) { return; }
160 	
161 	if ( BGAME.activeScreen.rframe % 16 === i % 16 ) {
162 		o.far = (o.sqDis && KFC.vec3.sqrDis(camera.position, o.pos) > o.sqDis * BGAME.renderDis * 0.1);
163 	}
164 	
165 	if ( o.far ) { return; }	// distance filtering
166 	
167 	if (this.activeShader !== o.shader) {
168 		gl.useProgram(o.shader.program);
169 		this.activeShader = o.shader;
170 		gl.uniformMatrix4fv(o.shader.pMatrix, false, camera.getMatrix());
171 	}
172 	if ( camera.dirty === true ) {
173 		gl.uniformMatrix4fv(o.shader.pMatrix, false, camera.getMatrix());
174 	}
175 	
176 	gl.uniformMatrix4fv(o.shader.mvMatrix, false, o.getMVMatrix());
177 	if ( o.shader.nMatrix ) {
178 		gl.uniformMatrix3fv(o.shader.nMatrix, false, o.getNMatrix());
179 	}
180 	
181 	for ( j = 0; j < o.shader.unis; j++ ) {
182 		
183 		u = o.shader.uniIds[j];
184 		l = o.shader.uniLocs[j];
185 		v = o.uniforms[u] || o.mesh.uniforms[u] || scene.uniforms[u] || camera.uniforms[u];
186 		if ( v === undefined ) {
187 			throw new Error('Object has undefined uniform: "' + KFC.UniStr[u] + '", "' + o.mesh.url_ +'"');
188 		}
189 		
190 		switch (KFC.UniTyp[u]) {
191 			case 10 :
192 				gl.uniform3fv(l, v);
193 				break;
194 			case 11 :
195 				gl.uniform4fv(l, v);
196 				break;
197 			case 0 :
198 				gl.uniform1f(l, v);
199 				break;
200 			case 4 :
201 				gl.uniform1i(l, v);
202 				break;
203 			default:
204 			throw new Error('Bad unifrom type: ' + KFC.UniStr[u] + ', ' + KFC.UniTyp[u] );			
205 		}
206 	}
207 	
208 	if ( o.culling === false ) {
209 		if ( this.culling === true ) {
210 			this.culling = false;
211 			gl.disable(gl.CULL_FACE);
212 		}
213 	} else {
214 		if ( this.culling === false ) {
215 			this.culling = true;
216 			gl.enable(gl.CULL_FACE);
217 		}
218 	}
219 	
220 	if ( o.blending === false ) {
221 		if ( this.blending === true ) {
222 			this.blending = false;
223 			gl.disable(gl.BLEND);
224 		}
225 	} else {
226 		if ( this.blending === false ) {
227 			this.blending = true;
228 			gl.enable(gl.BLEND);
229 		}
230 	}
231 	
232 	//BGAME.rendered++;
233 	
234 	o.mesh._render(gl, o.shader, o);
235 };

237 /**
238  * Renders depth map of all objects in a scene
239  * @param {KFC.Scene} scene Scene to get objects from
240  * @param {KFC.Camera} camera Camera to get perpective matrix from
241  */
242 KFC.CGLC.prototype.renderDepth = function(scene, camera) {
243 	
244 	var gl = this.gl,
245 		dsh = this.getShader('depth'),
246 		o, i;
247 		
248 	if ( !dsh.ready ) { return; }
249 	
250 	gl.useProgram(dsh.program);
251 	this.activeShader = dsh;
252 	gl.uniformMatrix4fv(dsh.pMatrix, false, camera.getMatrix());
253 	
254 	for (i = 0; i < scene.objects.length; i++) {
255 		o = scene.objects[i];
256 		
257 		if ( !o.draw ) { continue; }		
258 		if ( !o.mesh.ready ) { continue; }
259 		if ( !o.mesh._renderDepth ) { continue; }
260 		if ( o.far ) { return; }	// distance filtering
261 		
262 		gl.uniformMatrix4fv(dsh.mvMatrix, false, o.getMVMatrix());
263 		BGAME.rendered++;
264 		o.mesh._renderDepth(gl, dsh, o);
265 	}
266 	for (i = 0; i < scene.endObjects.length; i++) {
267 		o = scene.endObjects[i];
268 		
269 		if ( !o.draw ) { continue; }	
270 		if ( !o.mesh.ready ) { continue; }
271 		if ( !o.mesh._renderDepth ) { continue; }
272 		if ( o.far ) { return; }	// distance filtering
273 		
274 		gl.uniformMatrix4fv(dsh.mvMatrix, false, o.getMVMatrix());
275 		BGAME.rendered++;
276 		o.mesh._renderDepth(gl, dsh, o);
277 	}
278 };

281 /**
282  * Complete framebuffer object
283  * @constructor
284  * @param {!WebGLRenderingContext} gl WebGL rendering context
285  * @param {!number} width Famebuffer width
286  * @param {!number} height Framebuffer height
287  * @param {!boolean=} opt_type_float If true use gl.FLOAT as framebuffer texture format, if false or undefined use gl.UNSIGNED_INT
288  */
289 KFC.Framebuffer = function(gl, width, height, opt_type_float) {
290 	/** @type {!WebGLRenderingContext} */
291 	this.gl = gl;
292 	/** @type {WebGLFramebuffer} */
293 	this.fb = gl.createFramebuffer();
294 	/** @type {WebGLTexture} */
295 	this.tex = gl.createTexture();
296 	/** @type {WebGLRenderbuffer} */
297 	this.rb = gl.createRenderbuffer();
298 	
299 	this.width = width;
300 	this.height = height;
301 	
302 	this.type = (opt_type_float === true) ? gl.FLOAT : gl.UNSIGNED_BYTE;
303 	
304 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.fb);
305 	
306     gl.bindTexture(gl.TEXTURE_2D, this.tex);
307     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
308     gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
309 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
310 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
311 	
312     gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, this.width, this.height, 0, gl.RGBA, this.type, null);
313     gl.bindRenderbuffer(gl.RENDERBUFFER, this.rb);
314     gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16, this.width, this.height);
315 	
316     gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, this.tex, 0);
317     gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, this.rb);
318 	
319     gl.bindTexture(gl.TEXTURE_2D, null);
320     gl.bindRenderbuffer(gl.RENDERBUFFER, null);
321     gl.bindFramebuffer(gl.FRAMEBUFFER, null);
322 	
323 };

325 /**
326  * Gets framebuffer object, allocates one on GPU if does not exists
327  * @param {!string} name Name for framebuffer object
328  * @param {boolean=} opt_type_float Set to true if framebuffer needs to be of FLOAT type, default false UNSINGED_INT
329  * @param {number=} opt_width Width of framebuffer
330  * @param {number=} opt_height Height of framebuffer
331  * @return {!KFC.Framebuffer} Named framebuffer object
332  */
333 KFC.CGLC.prototype.getFramebuffer = function(name, opt_type_float, opt_width, opt_height) {
334 	
335 	var width = opt_width || this.width;
336 	var height = opt_height || this.height;
337 	
338 	if ( !this.framebuffers[name] ) {
339 		this.framebuffers[name] = new KFC.Framebuffer(this.gl, width, height, opt_type_float);	
340 	}
341 	
342 	return this.framebuffers[name];
343 };

345 /**
346  * Resizes all framebuffers allocated to this context
347  */
348 KFC.CGLC.prototype.resizeFrameBuffers = function() {
349 	if ( !this.fbResize ) { return; }
350 	
351 	var fb, n, gl = this.gl;
352 	for ( n in this.framebuffers ) {
353 		if ( this.framebuffers.hasOwnProperty(n) ) {
354 			fb = this.framebuffers[n];
355 			fb.width = this.width;
356 			fb.height = this.height;
357 			
358 			gl.bindFramebuffer(gl.FRAMEBUFFER, fb.fb);
359 			gl.bindTexture(gl.TEXTURE_2D, fb.tex);
360 			gl.bindRenderbuffer(gl.RENDERBUFFER, fb.rb);
361 			
362 			gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, fb.width, fb.height, 0, gl.RGBA, fb.type, null);
363 			gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16, fb.width, fb.height);
364 		}
365 	}
366 	
367 	gl.bindTexture(gl.TEXTURE_2D, null);
368     gl.bindRenderbuffer(gl.RENDERBUFFER, null);
369     gl.bindFramebuffer(gl.FRAMEBUFFER, null);
370 	
371 	this.fbResize = false;
372 };

426 /**
427  * Blurs source framebuffer according to depth buffer. Making field of view effect
428  * @param {KFC.Framebuffer} sourceFB Color framebuffer to blurr
429  * @param {KFC.Framebuffer} depthFB Depth framebuffer
430  * @param {?KFC.Framebuffer} destinationFb Framebuffer to render result to, set null to render to default framebuffer
431  * @param {number} [fBlur] Blur factor
432  * @param {number} [fDis] Focus distance
433  */
434 KFC.CGLC.prototype.gaussianBlur = function(sourceFB, depthFB, destinationFb, fBlur, fDis) {
435 	
436 	var vsh = this.getShader('vgauss'),
437 		hsh = this.getShader('hgauss'),
438 		tfb = this.getFramebuffer('tgauss'),
439 		gl = this.gl,
440 		blur = (typeof(fBlur) === 'number') ? fBlur : 1.0 / 512.0,
441 		focus = (typeof(fDis) === 'number') ? fDis : 0.25;
442 	
443 	if ( !vsh.ready || !hsh.ready ) { return; }	
444 	
445     gl.bindFramebuffer(gl.FRAMEBUFFER, tfb.fb);
446 	
447 	gl.useProgram(vsh.program);
448 	this.activeShader = vsh;
449 	
450 	vsh.attribPointer('aVerPosition', this.bScreenVerPos, 3);
451 	
452 	vsh.bindTexture(0, sourceFB.tex);
453 	vsh.bindTexture(1, depthFB.tex);
454 	
455 	gl.uniform1i(vsh.uniLocs[vsh.uniIds.indexOf(KFC.UniId.uTex0)], 0);
456 	gl.uniform1i(vsh.uniLocs[vsh.uniIds.indexOf(KFC.UniId.uTex1)], 1);
457 	gl.uniform1f(vsh.uniLocs[vsh.uniIds.indexOf(KFC.UniId.uBlur)], blur);
458 	gl.uniform1f(vsh.uniLocs[vsh.uniIds.indexOf(KFC.UniId.uFocus)], focus);
459 	
460 	gl.disable(gl.DEPTH_TEST);
461 	
462 	gl.drawArrays(gl.TRIANGLES, 0, 6);		
463 	
464     gl.bindFramebuffer(gl.FRAMEBUFFER, (destinationFb) ? destinationFb.fb : null);
465 		
466 	gl.useProgram(hsh.program);
467 	this.activeShader = hsh;
468 	
469 	hsh.attribPointer('aVerPosition', this.bScreenVerPos, 3);
470 	
471 	gl.activeTexture(gl.TEXTURE0);
472 	gl.bindTexture(gl.TEXTURE_2D, tfb.tex);
473 	gl.uniform1i(hsh.uniLocs[hsh.uniIds.indexOf(KFC.UniId.uTex0)], 0);
474 	gl.uniform1i(hsh.uniLocs[hsh.uniIds.indexOf(KFC.UniId.uTex1)], 1);
475 	gl.uniform1f(hsh.uniLocs[hsh.uniIds.indexOf(KFC.UniId.uBlur)], blur);
476 	gl.uniform1f(hsh.uniLocs[hsh.uniIds.indexOf(KFC.UniId.uFocus)], focus);
477 	
478 	
479 	gl.drawArrays(gl.TRIANGLES, 0, 6);
480 	
481 	gl.enable(gl.DEPTH_TEST);	
482 	gl.bindTexture(gl.TEXTURE_2D, null);						
483 	
484 };

511 /**
512  * Preloads shader, calls callback when shader is ready
513  * @param {string} name Predefined shader name
514  * @param {function()=} opt_callback Function to call when shader is ready
515  */
516 KFC.CGLC.prototype.readyShader = function( name, opt_callback ) {
517 	
518 	if ( typeof(name) !== 'string' ) { throw 'Wrong shader name!'; }
519 	
520 	if ( typeof(this.shaders[name]) !== 'undefined' ) {
521 		if ( opt_callback ) {
522 			opt_callback();
523 		}
524 		return;
525 	}
526 	
527 	switch (name) {
528 	case 'color' : 
529 		this.shaders[name] = new KFC.CustomMaterial( this, // shader for lines
530 							'color.vsh', 'color.fsh',
531 							['aVerPosition'],
532 							[KFC.UniId.uColor],
533 							opt_callback );
534 		break;
535 	case 'normal' : 
536 		this.shaders[name] = new KFC.CustomMaterial( this, // dunno what this is used for
537 							'normal.vsh', 'normal.fsh',
538 							['aVerPosition', 'aVerNormal'],							
539 							[KFC.UniId.uAmbientColor,
540 							 KFC.UniId.uDirColor,
541 							 KFC.UniId.uLightDirection,
542 							 KFC.UniId.uColor],
543 							opt_callback );
544 		break;
545 	case 'car' : 
546 		this.shaders[name] = new KFC.CustomMaterial( this,	// for rendering cars
547 							'car.vsh', 'car.fsh',
548 							['aVerPosition', 'aVerNormal', 'aTexCoord'],
549 							[/*'uAmbientColor', 'uDirColor',*/ 'uLightDirection', /*'uColor',*/ 'uTex0', 'uCamPosition'], opt_callback );
550 		break;
551 	case 'terrain' : 
552 		this.shaders[name] = new KFC.CustomMaterial( this,	// special terrain shader
553 							'terrain.vsh', 'terrain.fsh',
554 							['aVerPosition', 'aVerNormal', 'aTexCoord'],
555 							[KFC.UniId.uLightDirection,
556 							 KFC.UniId.uFog,
557 							 KFC.UniId.uScale,
558 							 KFC.UniId.uTex0],							 
559 							opt_callback );	
560 		break;
561 	case 'transparent' : 
562 		this.shaders[name] = new KFC.CustomMaterial( this,	// special terrain shader
563 							'transparent.vsh', 'transparent.fsh',
564 							['aVerPosition', 'aVerNormal', 'aTexCoord'],
565 							[KFC.UniId.uLightDirection,
566 							 KFC.UniId.uFog,
567 							 KFC.UniId.uScale,
568 							 KFC.UniId.uTex0],
569 							opt_callback );	
570 		break;
571 	case 'water' : 
572 		this.shaders[name] = new KFC.CustomMaterial( this,	// special terrain shader
573 							'water.vsh', 'water.fsh',
574 							['aVerPosition', 'aVerNormal', 'aTexCoord'],
575 							[/*'uAmbientColor', 'uDirColor',*/ 'uCamPosition', 'uLightDirection', 'uFog', 'uTime', 'uTex0'], opt_callback );	
576 		break;
577 	case 'multicol' : 
578 		this.shaders[name] = new KFC.CustomMaterial( this,	// special terrain shader
579 							'multicol.vsh', 'multicol.fsh',
580 							['aVerPosition', 'aVerNormal', 'aTexCoord'],
581 							[KFC.UniId.uLightDirection,
582 							 KFC.UniId.uFog,
583 							 KFC.UniId.uTex0,
584 							 KFC.UniId.uTex1,
585 							 KFC.UniId.uTex2,
586 							 KFC.UniId.uTex3,
587 							 KFC.UniId.uCamPosition], opt_callback );

742  * @constructor
743  * @param {!WebGLRenderingContext} gl WebGL rendering context

746  * @param {boolean=} opt_mipmap wheater texture should use mipmaps
747  */
748 KFC.Texture = function (gl, url, opt_callback, opt_mipmap) {
749 	var that = this;
750 	this.gl = gl;
751 	this.url = url;
752 	this._callback = opt_callback || null;

759 	};

762 	};
763 	
764 	this.img.src = this.url;
765 
766 	this.ready = false;

769 /**
770  * Copies loaded texture to GPU memmory
771  */
772 KFC.Texture.prototype._handleLoaded = function () {
773 	var gl = this.gl;
774 	gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, 1);
775 	gl.bindTexture(gl.TEXTURE_2D, this.tex);
776 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
777 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, (this.mipmap === true) ? gl.LINEAR_MIPMAP_NEAREST : gl.LINEAR);
778 	gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, this.img);
779 	
780 	if (this.mipmap === true) {
781 		gl.generateMipmap(gl.TEXTURE_2D);
782 	}
783 	gl.bindTexture(gl.TEXTURE_2D, null);
784 	
785 	this.ready = true;
786 	
787 	if (this._callback !== null) { 
788 		this._callback();
789 		this._callback = null;
790 	}
791 };

795  * @constructor

797  * @param {string} vsh_url Vertex shader source url
798  * @param {string} fsh_url Fragment shader source url
799  * @param {Array.<string|number>} att Shader attributes array

801  * @param {function()=} opt_callback function to call when texture is loaded
802  */
803 KFC.CustomMaterial = function (glc, vsh_url, fsh_url, att, uni, opt_callback) {
804 
805 	/** @type {KFC.CGLC} */
806 	this._glc = glc;
807 	/** @type {!WebGLRenderingContext} */
808 	this._gl = glc.gl;
809 	/** @type {string} */
810 	this._vsh_url = vsh_url;
811 	/** @type {string} */
812 	this._fsh_url = fsh_url;
813 	/** @type {Array.<string|number>} */
814 	this._att = att;
815 	/** @type {Array.<string|number>} */
816 	this._uni = uni;
817 	/** @type {?function()} */
818 	this._callback = opt_callback || null;

826 	
827 	this.unis = 0;
828 	this.uniIds = [];
829 	this.uniLocs = [];
830 	
831 	this._getShader(this._gl.VERTEX_SHADER, 'shaders/' + vsh_url);
832 	this._getShader(this._gl.FRAGMENT_SHADER, 'shaders/' + fsh_url);

835 /**
836  * Loads shader files
837  * @param {number} type Shader type gl.VERTEX_SHADER or gl.FRAGMENT_SHADER
838  * @param {string} url Shader url adress
839  */
840 KFC.CustomMaterial.prototype._getShader = function (type, url) {
841 	var xhr,
842 		that = this;
843 		
844 	xhr = new XMLHttpRequest();
845 	xhr.open("GET", url);
846 	xhr.onreadystatechange = function () {
847 		if (xhr.readyState === 4) {
848 			that._initShader(type, xhr.responseText);
849 		}
850 	};
851 	xhr.send();
852 };

854 /**
855  * Initializes loaded shader
856  * @param {number} type Shader type gl.VERTEX_SHADER or gl.FRAGMENT_SHADER
857  * @param {string} source Shader source code
858  */
859 KFC.CustomMaterial.prototype._initShader = function (type, source)
860 {
861 	var gl = this._gl,
862 		shader = gl.createShader(type);
863 
864 	gl.shaderSource(shader, source);
865 	gl.compileShader(shader);
866 	if ( !gl.getShaderParameter(shader, gl.COMPILE_STATUS) ) {
867 		gl.deleteShader(shader);
868 		throw new Error('ERROR: CGLC:initShaders - failed to compile shader: "' +
869 						((type === gl.VERTEX_SHADER) ? this._vsh_url : this._fsh_url) +
870 						'"\n' + gl.getShaderInfoLog(shader));
871 	}
872 	
873 	if (type === gl.VERTEX_SHADER) { this._vsh = shader; }
874 	if (type === gl.FRAGMENT_SHADER) { this._fsh = shader; }
875 	
876 	this._initProgram();
877 };
878 
879 /**
880  * Initializes shader program
881  */
882 KFC.CustomMaterial.prototype._initProgram = function ()
883 {    
884 	if ( this._vsh === null || this._fsh === null ) {
885 		return;
886 	}
887 	
888 	var a, u, name, pos, i, id,
889 		gl = this._gl;
890 		
891 	gl.attachShader(this.program, this._vsh);
892 	gl.attachShader(this.program, this._fsh);
893 	gl.linkProgram(this.program);
894 	
895 	if (!gl.getProgramParameter(this.program, gl.LINK_STATUS)) {
896 		console.error("ERROR: Could not initialise shaders!");
897 		return;
898 	}
899 	
900 	for ( i = 0; i < this._att.length; i++ ) {
901 		name = this._att[i].toString();
902 		pos = this.attributes[name] = gl.getAttribLocation(this.program, name);
903 		if ( pos !== -1 ) {
904 			gl.enableVertexAttribArray(this.attributes[name]);	// enable attribute	
905 			this.attEn[pos] = true;			// mark attribte as enabled
906 			//this.attBP[pos] = {};
907 			//this.attBP[pos].buffer = null;
908 			//this.attBP[pos].size = 0;
909 		} else {
910 			console.error("ERROR: Could not get attribute location \"" + name + "\"");
911 		}
912 	}
913 	
914 	for ( i = 0; i < this._uni.length; i++ ) {
915 		id = this._uni[i];
916 		
917 		if ( typeof id === 'string' ) {
918 			
919 			id = KFC.UniStr.indexOf(id);
920 			if ( id === -1 ) {
921 				throw new Error('shader uniform definition by string "' + this._uni[i] +  '" was not found! ' + this._vsh_url);
922 			}
923 		}
924 		if ( id === undefined ) {
925 			throw new Error('shader uniform ' + (i+1) +  ' id is undefined! ' + this._vsh_url);
926 		}
927 		
928 		name = KFC.UniStr[id];
929 		this.uniIds.push(id); 
930 		this.uniLocs.push(gl.getUniformLocation(this.program, name));		
931 		if ( this.uniLocs[this.unis] === null ) {
932 			console.log('WARNING: shader program "' + this._vsh_url + '" has undefined variable "' + name + '" ' + id);
933 		}		
934 		this.unis++;
935 	}
936 	
937 	this.pMatrix = gl.getUniformLocation(this.program, 'uPMatrix');
938 	this.mvMatrix = gl.getUniformLocation(this.program, 'uMVMatrix');
939 	this.nMatrix = gl.getUniformLocation(this.program, 'uNMatrix');
940 	
941 	for (u in this.attributes ) {
942 		if ( this.attributes[u] === null ) {
943 			console.log('WARNING: shader program has undefined variable "' + u + '"');
944 		}
945 	}
946 	
947 	this.ready = true;
948 	if (typeof this._callback === 'function') { 
949 		this._callback();
950 		this._callback = null;
951 	}
952 	
953 	return;
954 };

1097 /**
1098  * Calculates perspective matrix from set position and rotation
1099  */
1100 KFC.PerspectiveCamera.prototype._makeMatrix = function () {  
1101 	var position = this.position,
1102 		rotation = this.rotation;
1103 		
1104 	mat4.perspective(this.fov, this.aspect, this.near, this.far, this.pMatrix);
1105 	mat4.rotate(this.pMatrix, -rotation[0], [1, 0, 0]);
1106 	mat4.rotate(this.pMatrix, -rotation[1], [0, 1, 0]);
1107 	mat4.rotate(this.pMatrix, -rotation[2], [0, 0, 1]);
1108 	vec3.scale(position, -1, this.npos);
1109 	mat4.translate(this.pMatrix, this.npos);
1110 	
1111 	this.uniforms[KFC.UniId.uCamPosition] = [position[0], position[1], position[2]];
1112 	
1113 	this.dirty = false;
1114 };

1124 	if ( p[0] === x && p[1] === y && p[2] === z ) { return; }
1125 	p[0] = x;
1126 	p[1] = y;
1127 	p[2] = z;

1146 /**
1147  * Rotates camera to face at specific point
1148  * @param {number} lx X coordinate
1149  * @param {number} ly Y coordinate
1150  * @param {number} lz Z coordinate
1151  */
1152 KFC.PerspectiveCamera.prototype.lookAt = function (lx, ly, lz) {
1153 	var dx = lx - this.position[0],
1154 		dy = ly - this.position[1],
1155 		dz = lz - this.position[2],
1156 		vrot, hrot;
1157 		
1158 	hrot = Math.atan(dx/dz);
1159 	if ( dz > 0 ) { hrot += Math.PI; }
1160 	vrot = Math.atan(dy/Math.sqrt(dx*dx+dz*dz));
1161 	this.setRotation(vrot, hrot, 0);
1162 };

1164 /**
1165  * moves (floats) camera to specific point
1166  * @param {vec3} pos_wc Follow point location
1167  * @param {number} iStep Integration step.
1168  *
1169  */
1170 KFC.PerspectiveCamera.prototype.follow = function ( pos_wc, iStep ) {
1171 	
1172 	var f = 0,
1173 		d = 0;
1174 		
1175 	d = vec3.dist(this.position, pos_wc);		
1176 	f = 40 * (d - 7);
1177 	vec3.direction(this.position, pos_wc, this.tv);
1178 	KFC.vec3.addscaled(this.vel, this.tv, -f * iStep );
1179 	vec3.scale(this.vel, 0.98);	
1180 	
1181 	KFC.vec3.addscaled(this.position, this.vel, iStep );	
1182 	this.position[1] = pos_wc[1] + 3;
1183 	
1184 	this.dirty = true;
1185 };

1188 /**
1189  * moves (floats) camera to specific point
1190  * @param {vec3} pos_wc Follow point location
1191  * @param {number} iStep Integration step.
1192  *
1193  */
1194 KFC.PerspectiveCamera.prototype.goTo = function ( pos_wc, iStep ) {
1195 	
1196 	var f = 0;
1197 		
1198 	f = vec3.dist(this.position, pos_wc) * 20;
1199 	vec3.direction(this.position, pos_wc, this.tv);
1200 	KFC.vec3.addscaled(this.vel, this.tv, -f * iStep );
